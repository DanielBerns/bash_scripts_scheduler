Here is the updated `Makefile` including the new `backup` target. I also updated the `help` section so the new command is properly documented when you run `make`.

### Updated `Makefile`

```makefile
# Makefile for YAML Cron Scheduler

# Variables
DEST_DIR = ~/Commands
SCRIPT = ./scheduler.py
CONFIG = schedule.yaml
LOG_FILE = schedule.log
ASSETS_DIR ?= ./assets

.PHONY: help init install backup sync clear status logs

# Default target when just typing 'make'
help:
	@echo "Usage: make [target]"
	@echo ""
	@echo "Targets:"
	@echo "  init      Create $(DEST_DIR), copy dummy_script.sh and schedule.yaml from ASSETS_DIR"
	@echo "  install   Install required Python dependencies using uv"
	@echo "  backup    Manually backup the entire current crontab to $(DEST_DIR)/crontab_backups"
	@echo "  sync      Backup crontab, then apply the jobs defined in $(CONFIG)"
	@echo "  clear     Remove all managed jobs from crontab"
	@echo "  status    List all currently managed jobs in crontab"
	@echo "  logs      Tail the execution log file ($(LOG_FILE))"

init:
	@echo "Creating $(DEST_DIR) if it does not exist..."
	mkdir -p $(DEST_DIR)
	@echo "Copying assets from $(ASSETS_DIR)..."
	cp $(ASSETS_DIR)/dummy_script.sh $(DEST_DIR)/
	cp $(ASSETS_DIR)/schedule.yaml $(DEST_DIR)/
	@echo "Setting execution permissions for the dummy script..."
	chmod +x $(DEST_DIR)/dummy_script.sh
	@echo "Initialization complete. You can now run 'make sync' from $(DEST_DIR)."

install:
	uv pip install pyyaml python-crontab

backup:
	$(SCRIPT) backup --dir $(DEST_DIR)

sync:
	$(SCRIPT) --config $(CONFIG) sync

clear:
	$(SCRIPT) --config $(CONFIG) clear

status:
	$(SCRIPT) --config $(CONFIG) status

logs:
	tail -f $(LOG_FILE)

```

Now you can run `make backup` at any time to take a manual snapshot, and `make sync` will automatically trigger that same backup logic right before it applies your YAML configuration.
